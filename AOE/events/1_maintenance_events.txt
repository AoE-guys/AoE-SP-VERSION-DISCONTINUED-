####Game-Master and CleanUp

country_event = {
	id = 6016818 #Game Master Maintenance 
	title = "Game Master"
	desc = "Game Master"

	trigger = { 
	           ai = yes
			   month = 1
			   NOT = { month = 2 }
			   NOT = { has_country_modifier = gm_timer }
	}

	mean_time_to_happen = {
		days = 1
	}		
	
	option = {
		name = "Game Master"
		any_country = { limit = { has_country_flag = game_master } clr_country_flag = game_master }
		random_country = {
			limit = { AND = { exists = yes ai = yes } }
			set_country_flag = game_master
		}
		any_country = {
			#limit = { ai = yes }
			add_country_modifier = { name = gm_timer  duration = 60 }
		}
		add_country_modifier = { name = gm_timer  duration = 60 }
		
		any_country = {
			limit = { exists = no OR = { is_mobilised = yes total_amount_of_ships = 1 } }
			country_event = 6016821
		}
		any_country = {
			limit = { exists = no has_country_flag = dynamic_tag }
			country_event = 9999817
		}
		any_country = {
			limit = { exists = no NOT = { is_cultural_union = yes } any_core = { owner = { has_country_flag = dynamic_tag } } }
			country_event = 9999810
		}
		any_country = {
			limit = { exists = yes has_country_flag = dynamic_tag }
			country_event = 9999807
		}
	}
}
